### TinyMamba (مامبا مكتوب بـ PyTorch صافٍ)

> **لماذا نكتبه بأنفسنا؟** مكتبة `mamba-ssm` الرسمية تحتاج CUDA وتفشل
> على CPU. نستخدم نسخة بسيطة تعيد نفس المبدأ.

**المعمارية الكاملة:**

```python
class TinyMamba(nn.Module):
    """
    نسخة مبسّطة من Mamba بـ PyTorch صافٍ تعمل على CPU
    المرجع: ورقة Mamba (Gu & Dao, 2023) — الصيغة الأساسية فقط
    """
    def __init__(self, vocab_size, embed_dim=128, state_dim=64, num_layers=2):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        self.layers = nn.ModuleList([
            MambaBlock(embed_dim, state_dim) for _ in range(num_layers)
        ])
        self.norm = nn.LayerNorm(embed_dim)
        self.head = nn.Linear(embed_dim, vocab_size)

class MambaBlock(nn.Module):
    """
    الفكرة الأساسية لـ Mamba: SSM انتقائي (Selective SSM)
    الفرق عن SSM العادي: B وC تعتمد على المدخل x (انتقائية)
    """
    def __init__(self, embed_dim, state_dim):
        super().__init__()
        self.d_inner = embed_dim * 2   # توسيع داخلي
        self.dt_rank = max(embed_dim // 16, 1)
        
        # إسقاطات المدخل
        self.in_proj = nn.Linear(embed_dim, self.d_inner * 2, bias=False)
        
        # معاملات SSM الانتقائية (تعتمد على المدخل)
        self.x_proj = nn.Linear(self.d_inner, self.dt_rank + state_dim * 2, bias=False)
        self.dt_proj = nn.Linear(self.dt_rank, self.d_inner)
        
        # المعامل A (لوغاريتمي لضمان الاستقرار)
        A = torch.arange(1, state_dim + 1).float()
        self.A_log = nn.Parameter(torch.log(A).unsqueeze(0).expand(self.d_inner, -1))
        self.D = nn.Parameter(torch.ones(self.d_inner))
        
        # إسقاط الخرج
        self.out_proj = nn.Linear(self.d_inner, embed_dim, bias=False)
        self.norm = nn.LayerNorm(embed_dim)
    
    def forward(self, x):
        B, T, D = x.shape
        residual = x
        x = self.norm(x)
        
        # توسيع الدخل
        xz = self.in_proj(x)    # (B, T, 2*d_inner)
        x_in, z = xz.chunk(2, dim=-1)
        
        # حساب معاملات SSM الانتقائية
        x_dbl = self.x_proj(x_in)   # (B, T, dt_rank + 2*N)
        dt_raw, B_t, C_t = torch.split(
            x_dbl,
            [self.dt_rank, self.A_log.shape[-1], self.A_log.shape[-1]],
            dim=-1
        )
        dt = torch.nn.functional.softplus(self.dt_proj(dt_raw))  # (B, T, d_inner) > 0
        A = -torch.exp(self.A_log.float())  # (d_inner, N) < 0 للاستقرار
        
        # SSM انتقائي — حلقة بسيطة على CPU
        h = torch.zeros(B, self.A_log.shape[0], self.A_log.shape[1], device=x.device)
        ys = []
        for t in range(T):
            # تحديث الحالة: انتقائي لأن dt وB وC تتغيّر مع المدخل
            h = h * torch.exp(dt[:, t, :, None] * A[None, :, :])  + \
                B_t[:, t, None, :] * x_in[:, t, :, None]
            y_t = (h * C_t[:, t, None, :]).sum(dim=-1)  # (B, d_inner)
            ys.append(y_t)
        
        y = torch.stack(ys, dim=1)  # (B, T, d_inner)
        y = y + self.D[None, None, :] * x_in   # skip connection
        y = y * torch.nn.functional.silu(z)     # بوابة SiLU
        
        return self.out_proj(y) + residual
```

### TinyTransformer (ترانسفورمر بسيط)

```python
class TinyTransformer(nn.Module):
    """
    ترانسفورمر بسيط: Embedding + Attention + FFN
    الترموذج التقليدي للمقارنة الأساسية
    """
    def __init__(self, vocab_size, embed_dim=128, num_heads=4,
                 num_layers=2, ff_dim=256, max_seq=512):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        self.pos_emb = nn.Embedding(max_seq, embed_dim)  # ترميز موضعي بسيط
        self.layers = nn.ModuleList([
            TransformerBlock(embed_dim, num_heads, ff_dim)
            for _ in range(num_layers)
        ])
        self.norm = nn.LayerNorm(embed_dim)
        self.head = nn.Linear(embed_dim, vocab_size)

class TransformerBlock(nn.Module):
    """طبقة ترانسفورمر: Attention + FFN مع Residual"""
    def __init__(self, d, n_heads, ff_dim):
        super().__init__()
        self.attn = nn.MultiheadAttention(d, n_heads, batch_first=True)
        self.ff = nn.Sequential(
            nn.Linear(d, ff_dim), nn.GELU(), nn.Linear(ff_dim, d)
        )
        self.norm1 = nn.LayerNorm(d)
        self.norm2 = nn.LayerNorm(d)
    
    def forward(self, x):
        # Attention مع قناع سببي (causal mask — لا ينظر للأمام)
        B, T, D = x.shape
        mask = torch.triu(torch.full((T, T), float('-inf')), diagonal=1)
        attn_out, _ = self.attn(self.norm1(x), self.norm1(x), self.norm1(x),
                                attn_mask=mask)
        x = x + attn_out
        x = x + self.ff(self.norm2(x))
        return x
```

def verify_param_parity(wameed, mamba, transformer):
    w = sum(p.numel() for p in wameed.parameters())
    m = sum(p.numel() for p in mamba.parameters())
    t = sum(p.numel() for p in transformer.parameters())
    print(f"وميض: {w:,} | Mamba: {m:,} | Transformer: {t:,}")
    assert abs(w-m)/w < 0.05, f"فرق كبير بين وميض ومامبا: {abs(w-m)/w*100:.1f}%"
    assert abs(w-t)/w < 0.05, f"فرق كبير بين وميض والترانسفورمر: {abs(w-t)/w*100:.1f}%"
    print("✅ الأحجام متطابقة ضمن ±5%")

### المهمة 4.3 — قياس السرعة والرامة على جهازه

**قياس السرعة (الاستدلال):**

```python
import time, torch

def benchmark_inference(model, batch_size=1, seq_len=64, n_runs=100):
    """
    يقيس سرعة الاستدلال (tokens/second) على CPU
    نستخدم batch_size=1 seq_len=64 لمحاكاة الاستخدام الفعلي
    """
    model.eval()
    dummy = torch.randint(0, 2048, (batch_size, seq_len))
    
    # إحماء (warmup) أولاً
    for _ in range(5):
        with torch.no_grad(): _ = model(dummy)
    
    # القياس
    times = []
    for _ in range(n_runs):
        t0 = time.perf_counter()
        with torch.no_grad(): _ = model(dummy)
        t1 = time.perf_counter()
        times.append(t1 - t0)
    
    avg_time = np.mean(times)
    tokens_per_sec = (batch_size * seq_len) / avg_time
    print(f"متوسط الزمن: {avg_time*1000:.2f} مللي ثانية")
    print(f"السرعة: {tokens_per_sec:.1f} توكن/ثانية")
    return tokens_per_sec
```

def length_scaling_benchmark(models_dict, lengths=[64, 128, 256, 512, 1024]):
    """
    يرسم كيف تتصرّف النماذج مع نصوص أطول
    هنا يُفترض أن وميض ومامبا يتفوّقان على الترانسفورمر (الأخير O(n^2))

    results = {}
    for name, model in models_dict.items():
        times, rams = [], []
        for L in lengths:
            dummy = torch.randint(0, 2048, (1, L))

## 📊 تحليل النتيجة المتوقّعة (تقدير صادق)

| المحور | وميض المتوقّع | مامبا المتوقّع | ترانسفورمر المتوقّع |
|-------|-------------|-------------|-------------------|
| PPL الجودة | متوسط أو أفضل | متوسط | جيد (لكن يُعاني مع نصوص طويلة) |
| السرعة (CPU) | جيدة لكن أبطأ قليلاً من مامبا | سريع | بطيء جداً (O(n²)) |
| الرامة | أقل (بفضل الضغط) | متوسطة | أكبر (O(n²)) |
| النصوص الطويلة | **أفضل** (بفضل الذاكرة) | جيد | ضعيف |

> **الهدف الواقعي:** وميض يتفوّق في محور النصوص الطويلة + الرامة.
> لا نتوقّع تفوّقاً مطلقاً في كل المحاور — ولا نحتاجه لأهداف المشروع.
